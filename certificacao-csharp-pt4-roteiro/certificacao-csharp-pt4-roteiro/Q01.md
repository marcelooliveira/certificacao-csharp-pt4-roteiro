# RESPOSTA:

```
public string GetResponse(char letter)
{
    string response;
    switch(letter)
    {
        case 'a':
            response = "animal":
            break;
        case 'm':
            response = "mineral";
            break:
        default:
            response = "invalid choice";
            break:
    }
    return response;
}
```
